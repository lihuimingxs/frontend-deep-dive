并发控制

Promise.all(urls.map(fetch)) 最容易被忽略的风险是：并发度 = 数组长度。1000 个 URL 就是 1000 个同时进行的 fetch，浏览器限流、服务器过载、内存爆炸。本章给出 5 个生产级模式：信号量、动态池、重试退避、令牌桶、批处理，都配手写实现 + 取消支持。

阅读时长 ~22 分钟 前置 04 Promise 组合器 · 06 AbortController 对应 javascript.info Promise API

01 速览

5 种核心模式及对应工具：

模式 用途 工业库
Semaphore 最多 N 个并发持有"许可证" p-limit
Pool 动态池：完成一个补一个 p-map / p-pool
Queue FIFO 队列，按序处理 p-queue / bull
Token bucket / Rate limiter 限速率（N 次/秒） bottleneck / p-throttle
Retry + Backoff 失败重试 + 指数退避 p-retry / async-retry

02 底层逻辑

Semaphore（信号量）

核心思路：维护一个"待执行函数"队列 + 当前运行数。Promise.withResolvers（ES2024）让实现很干净：

javascript 
function pLimit(concurrency) {
  const queue = [];
  let active = 0;

  function next() {
    if (active >= concurrency || queue.length === 0) return;
    active++;
    const { fn, resolve, reject } = queue.shift();
    fn().then(resolve, reject).finally(() => {
      active--;
      next();
    });
  }

  return function (fn) {
    return new Promise((resolve, reject) => {
      queue.push({ fn, resolve, reject });
      next();
    });
  };
}

// 用法
const limit = pLimit(5);
const results = await Promise.all(
  urls.map(u => limit(() => fetch(u)))
);
// 最多 5 个 fetch 同时在跑，其余排队

动态池（mapConcurrent）

比 semaphore 更直接的 API：给一个 iterable + 处理函数 + 并发度，内部自动调度。

javascript 
async function mapConcurrent(items, fn, concurrency = 5) {
  const results = new Array(items.length);
  const executing = new Set();
  let index = 0;

  async function runOne() {
    const i = index++;
    if (i >= items.length) return;
    const p = Promise.resolve(fn(items[i], i)).then(v => { results[i] = v; });
    executing.add(p);
    await p;
    executing.delete(p);
    await runOne();
  }

  // 启动 N 条"执行链"
  await Promise.all(Array.from({ length: Math.min(concurrency, items.length) }, runOne));
  return results;
}

const results = await mapConcurrent(urls, (url) => fetch(url), 10);

Task Queue（带取消）

javascript 
class Queue {
  #active = 0;
  #pending = [];
  constructor(concurrency) { this.concurrency = concurrency; }

  add(fn, { signal } = {}) {
    return new Promise((resolve, reject) => {
      if (signal?.aborted) return reject(signal.reason);
      const task = { fn, resolve, reject, signal };
      this.#pending.push(task);
      signal?.addEventListener("abort", () => {
        const idx = this.#pending.indexOf(task);
        if (idx >= 0) this.#pending.splice(idx, 1);   // 从队列移除
        reject(signal.reason);
      }, { once: true });
      this.#run();
    });
  }

  async #run() {
    if (this.#active >= this.concurrency || !this.#pending.length) return;
    this.#active++;
    const { fn, resolve, reject, signal } = this.#pending.shift();
    try {
      if (signal?.aborted) reject(signal.reason);
      else resolve(await fn(signal));
    } catch (e) { reject(e); }
    this.#active--;
    this.#run();
  }
}

Token Bucket（速率限制）

javascript 
class RateLimiter {
  #lastRefill = Date.now();
  #tokens;
  constructor({ ratePerSec, burst = ratePerSec }) {
    this.rate = ratePerSec;
    this.burst = burst;
    this.#tokens = burst;
  }
  async acquire() {
    while (true) {
      const now = Date.now();
      const elapsed = (now - this.#lastRefill) / 1000;
      this.#tokens = Math.min(this.burst, this.#tokens + elapsed * this.rate);
      this.#lastRefill = now;
      if (this.#tokens >= 1) { this.#tokens--; return; }
      const wait = ((1 - this.#tokens) / this.rate) * 1000;
      await new Promise(r => setTimeout(r, wait));
    }
  }
}

const rl = new RateLimiter({ ratePerSec: 10, burst: 20 });
for (const url of urls) {
  await rl.acquire();
  void fetch(url);     // 10 req/s，突发允许 20
}

Retry + 指数退避 + Jitter

javascript 
async function retry(fn, {
  maxAttempts = 3,
  baseMs = 300,
  maxMs = 10_000,
  factor = 2,
  jitter = 0.3,        // 0-1 的抖动比例，防"重试雪崩"
  shouldRetry = (err) => err.name !== "AbortError",
  signal,
} = {}) {
  let lastErr;
  for (let i = 0; i < maxAttempts; i++) {
    signal?.throwIfAborted();
    try { return await fn(signal); }
    catch (err) {
      lastErr = err;
      if (!shouldRetry(err) || i === maxAttempts - 1) throw err;
      const base = Math.min(maxMs, baseMs * factor ** i);
      const delay = base * (1 + jitter * (Math.random() - 0.5));
      await new Promise((res, rej) => {
        const t = setTimeout(res, delay);
        signal?.addEventListener("abort", () => { clearTimeout(t); rej(signal.reason); }, { once: true });
      });
    }
  }
  throw new Error(`Failed after ${maxAttempts}`, { cause: lastErr });
}

批处理（Chunking）

javascript 
// 把大数组切成块，串行处理每块（块内并发）
async function batchProcess(items, fn, { chunkSize = 100, concurrency = 10 } = {}) {
  const out = [];
  for (let i = 0; i < items.length; i += chunkSize) {
    const chunk = items.slice(i, i + chunkSize);
    const chunkOut = await mapConcurrent(chunk, fn, concurrency);
    out.push(...chunkOut);
    // 可选：每批之间让行或休息
    await new Promise(r => setTimeout(r, 0));
  }
  return out;
}

// 适合场景：DB 批量写、服务端批 API 调用

03 应用场景

爬虫：限并发 + 限速 + 重试

javascript 
const limit = pLimit(5);
const rl = new RateLimiter({ ratePerSec: 10 });

async function crawl(url) {
  await rl.acquire();
  return retry(
    (signal) => fetch(url, { signal }),
    { maxAttempts: 3, signal: globalAbort.signal }
  );
}

const results = await Promise.all(urls.map(u => limit(() => crawl(u))));

文件批量上传

async function uploadFiles(files, { onProgress, signal } = {}) {
  let done = 0;
  const results = await mapConcurrent(
    files,

      const url = await retry(

Web Worker 任务池

javascript 
class WorkerPool {
  #workers = [];
  #queue = [];
  constructor(workerUrl, size = navigator.hardwareConcurrency ?? 4) {
    for (let i = 0; i < size; i++) this.#workers.push({ worker: new Worker(workerUrl), busy: false });
  }
  exec(task) {
    return new Promise((resolve, reject) => {
      const slot = this.#workers.find(w => !w.busy);
      if (!slot) { this.#queue.push({ task, resolve, reject }); return; }
      this.#dispatch(slot, task, resolve, reject);
    });
  }
  #dispatch(slot, task, resolve, reject) {
    slot.busy = true;
    slot.worker.onmessage = (e) => {
      slot.busy = false;
      resolve(e.data);
      const next = this.#queue.shift();
      if (next) this.#dispatch(slot, next.task, next.resolve, next.reject);
    };
    slot.worker.onerror = reject;
    slot.worker.postMessage(task);
  }
}

04 典型代码

Race Pool（动态补充）

javascript 
// 始终保持 N 个 Promise 在跑，完成一个立即补一个
async function* raceAll(iterable, concurrency) {
  const iter = iterable[Symbol.iterator]();
  const executing = new Map();    // Promise → index
  let idx = 0;
  function spawn() {
    const { value, done } = iter.next();
    if (done) return false;
    const cur = idx++;
    const p = Promise.resolve(value()).then(v => ({ v, p }));
    executing.set(p, cur);
    return true;
  }
  for (let i = 0; i < concurrency; i++) if (!spawn()) break;
  while (executing.size) {
    const { v, p } = await Promise.race(executing.keys());
    executing.delete(p);
    yield v;
    spawn();
  }
}

// 用法：按完成顺序消费结果
for await (const result of raceAll(urls.map(u => () => fetch(u)), 5)) {
  render(result);    // 完成一个就渲染，不等全部
}

异步管道（async pipeline）

javascript 
// 管道：取 → 处理 → 存
async function pipeline(source, stages, concurrency = 5) {
  let current = source;
  for (const stage of stages) {
    current = mapConcurrent(current, stage, concurrency);
  }
  return current;
}

const result = await pipeline(
  urls,
  [
    (url) => fetch(url).then(r => r.json()),
    (data) => transform(data),
    (data) => save(data),
  ],
  10
);

Settled 批处理（不让一个失败搞垮整体）

javascript 
async function mapConcurrentSettled(items, fn, concurrency = 5) {
  const limit = pLimit(concurrency);
  return Promise.allSettled(items.map(item => limit(() => fn(item))));
}

const results = await mapConcurrentSettled(
  items,
  (i) => syncToRemote(i),
  10
);
const failed = results.filter(r => r.status === "rejected");
logger.warn({ failed: failed.length });

异步 debounce（取消旧请求）

javascript 
function debounceAsync(fn, ms) {
  let timer, controller;
  return (...args) => {
    clearTimeout(timer);
    controller?.abort();
    controller = new AbortController();
    return new Promise((resolve, reject) => {
      timer = setTimeout(async () => {
        try { resolve(await fn({ signal: controller.signal }, ...args)); }
        catch (err) { if (err.name !== "AbortError") reject(err); }
      }, ms);
    });
  };
}

const debouncedSearch = debounceAsync(
  async ({ signal }, q) => fetch(`/search?q=${q}`, { signal }).then(r => r.json()),
  300
);
// 用户连续输入时，只有最后一次真的请求服务器
// 且前面"已经发出但未返回"的请求会被 abort

05 常见陷阱

Promise.all(map) 零并发控制

最常见的生产事故源头。1000 个 URL 就是 1000 并发请求 → 服务器 502 / 浏览器节流。默认加 pLimit 包一层。

队列无上限导致 OOM

生产者 > 消费者速度时，任务堆积在内存。生产级队列要加 maxPending —— 超出就 reject 或 block 生产者（背压）。

重试不区分可重试错误

4xx（如 400 Bad Request、401 Unauthorized）重试多少次都没用。shouldRetry 应只对 5xx / 网络错误返回 true。业务错误和 AbortError 不重试。

没有 jitter 的重试雪崩

服务下游挂了 → 大量客户端同时重试 → 恢复后第一波请求把服务再打挂。jitter（随机抖动）让重试时间分散。

限流器内部时钟与真实时间漂移

基于 Date.now() 的 rate limiter 在系统时钟跳变（NTP 校准）时会异常。用 performance.now() 或在窗口内计数（固定窗口 / 滑动窗口）更稳。

共享 state 的竞态

并发任务修改同一变量（counter、cache）时要用 await 序列化。let x = 0; await Promise.all([inc(), inc()]) 如果 inc 里 x = x + 1 且中间有 await，结果可能是 1 而不是 2。P1-14 讲过。

取消时的资源泄漏

pool 里的 Worker / 连接如果只在 resolve 路径释放，cancel 时就泄漏。finally 是朋友：try { ... } finally { release() }。AbortSignal 场景要 listener.abort 里清理。

错误不被计数的 semaphore

信号量实现里忘了在 catch 路径上 active-- → 一次失败永久占一个许可证 → 最终饿死。用 .finally(() => { active--; next(); }) 统一处理。
