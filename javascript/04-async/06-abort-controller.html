AbortController 与可取消异步

Promise 天生不可取消 —— Promise.race 只能"忽略"输家的结果，无法"停止"输家的执行。AbortController / AbortSignal 是 DOM 规范提供的"取消协议"，原生 fetch、setTimeout、addEventListener、流、Web Crypto 等都集成了它。本章把这套协议讲透 + 给出工程级可取消 async 模板。

阅读时长 ~18 分钟 前置 04 Promise 组合器 · 05 async/await 对应 MDN AbortController

01 速览

javascript 
const controller = new AbortController();
const signal = controller.signal;

// 1. 传给原生 API
fetch(url, { signal });                      // fetch 会在 abort 时 reject
addEventListener("click", fn, { signal });   // abort 时自动移除监听器
setTimeout(fn, 1000, { signal });            // Node 17+
navigator.locks.request("lock", { signal }, fn);

// 2. 2 秒后取消
setTimeout(() => controller.abort(new DOMException("User canceled", "AbortError")), 2000);

// 3. 检查状态
signal.aborted;     // false / true
signal.reason;      // abort(reason) 里传的值（默认 DOMException）

// 4. 辅助构造器（ES2024）
const timeoutSignal = AbortSignal.timeout(5000);          // 5s 后自动 abort
const combined = AbortSignal.any([s1, s2, timeoutSignal]); // 任一 abort 则整体 abort

原生支持 signal 的 API 清单

API 行为
fetch(url, { signal }) abort 时 reject 为 AbortError
Request 构造器 { signal } 同上
addEventListener(..., { signal }) abort 时自动 removeEventListener
EventTarget.prototype.addEventListener 同上
setTimeout(fn, ms, ...args)（Node 17+） 没有原生 signal 参数，但 Node timers/promises 有
setTimeout(signal) via timers/promises（Node） abort 时 reject
ReadableStream/WritableStream cancel/abort 带 reason
navigator.locks.request abort 取消等待
WritableStream.getWriter().abort(reason) 发起方关闭
Web Crypto 大量 API 签名/加密长时运算可取消

02 底层逻辑

取消协议的设计

AbortController 把"取消"分成两半：

  • 发起方（谁想取消）持有 controller.abort() 方法。
  • 被取消方（异步任务内部）观察 signal：signal.aborted 查状态、addEventListener("abort", ...) 监听、signal.throwIfAborted() 检查+抛错。

这种"外部事件 + 内部观察"的模式让调用方和被调用方解耦：fetch 不知道谁要取消它，controller 也不直接碰 fetch 内部。

Signal 的完整 API

成员 含义
signal.aborted boolean，当前是否已 abort
signal.reason abort 时传入的原因（默认 DOMException AbortError）
signal.throwIfAborted() 若 aborted 则抛 reason；否则 noop
signal.addEventListener("abort", fn) 监听；调用一次
signal.onabort = fn 属性形式（只能挂一个）
AbortSignal.abort(reason?)（静态） 返回一个已 aborted 的 signal（用于测试）
AbortSignal.timeout(ms)（ES2024） 返回 N ms 后自动 abort 的 signal
AbortSignal.any([signals])（ES2024） 任一 abort 则合成 signal 也 abort

一次性语义

signal 一旦 aborted，永远是 aborted。abort() 被调用多次只生效第一次，reason 固定为首次的值。"取消"是终态 —— 和 Promise 的 rejected 类似。

AbortSignal.timeout 的设计

javascript 
// 简化实现
AbortSignal.timeout = function (ms) {
  const controller = new AbortController();
  setTimeout(() => controller.abort(new DOMException("TimeoutError", "TimeoutError")), ms);
  return controller.signal;
};

// 用法：和 fetch 组合
const res = await fetch(url, { signal: AbortSignal.timeout(5000) });
// 5s 后 fetch reject with TimeoutError

// ⚠ Error name 是 "TimeoutError" 不是 "AbortError"
try { await fetch(url, { signal: AbortSignal.timeout(5000) }); }
catch (err) {
  if (err.name === "TimeoutError") /* 超时 */;
  else if (err.name === "AbortError") /* 手动取消 */;
  else /* 网络错误 */;
}

AbortSignal.any 合并信号

javascript 
// 需要"手动取消 OR 超时 OR 父组件卸载" 三个条件任一满足
const userCancel = new AbortController();
const combined = AbortSignal.any([
  userCancel.signal,
  AbortSignal.timeout(5000),
  parentAbortSignal,      // 外部传入的父级取消信号
]);

await fetch(url, { signal: combined });

// 老浏览器 polyfill：手动组合
function anyOf(...signals) {
  const c = new AbortController();
  for (const s of signals) {
    if (s.aborted) c.abort(s.reason);
    else s.addEventListener("abort", () => c.abort(s.reason), { once: true, signal: c.signal });
  }
  return c.signal;
}

在自己的 async 函数里支持 signal

三步：进入时检查 / 长时运算中周期检查 / 监听 abort 事件触发清理。

javascript 
async function longRun(data, { signal } = {}) {
  signal?.throwIfAborted();        // 1. 进入即查

  for (let i = 0; i < data.length; i++) {
    if (i % 100 === 0) signal?.throwIfAborted();   // 2. 周期查
    process(data[i]);
    if (i % 1000 === 0) await new Promise(r => setTimeout(r, 0));  // 让行
  }
}

// 带清理的版本
function pollUntil(predicate, { signal } = {}) {
  return new Promise((resolve, reject) => {
    let timer;
    function cleanup() {
      clearTimeout(timer);
      signal?.removeEventListener("abort", onAbort);
    }
    function onAbort() {
      cleanup();
      reject(signal.reason);
    }
    function tick() {
      if (signal?.aborted) return onAbort();
      if (predicate()) { cleanup(); resolve(); return; }
      timer = setTimeout(tick, 100);
    }
    signal?.addEventListener("abort", onAbort, { once: true });
    tick();
  });
}

03 应用场景

搜索输入：上次的请求自动取消

javascript 
let searchController = null;

async function onSearch(query) {
  searchController?.abort();               // 取消上次未完成的
  searchController = new AbortController();
  try {
    const res = await fetch(`/api/search?q=${encodeURIComponent(query)}`, {
      signal: searchController.signal,
    });
    render(await res.json());
  } catch (err) {
    if (err.name !== "AbortError") throw err;   // 只静默 abort
  }
}

input.addEventListener("input", (e) => onSearch(e.target.value));

React useEffect 清理

javascript 
useEffect(() => {
  const controller = new AbortController();
  (async () => {
    try {
      const res = await fetch(`/users/${id}`, { signal: controller.signal });
      setData(await res.json());
    } catch (err) {
      if (err.name !== "AbortError") setError(err);
    }
  })();
  return () => controller.abort();         // 组件卸载 / deps 变化时取消
}, [id]);

// 或者更现代：React 的 use hook + Suspense + cancellation

事件监听的"自动移除"

javascript 
function setupDragAndDrop(el) {
  const controller = new AbortController();
  const { signal } = controller;

  el.addEventListener("dragstart", onDragStart, { signal });
  document.addEventListener("dragover", onDragOver, { signal });
  document.addEventListener("drop", onDrop, { signal });
  document.addEventListener("dragend", onDragEnd, { signal });

  return () => controller.abort();         // 一次 abort 清除所有 listener
}

比手动 removeEventListener 4 次可靠得多。推荐所有新代码都用这种模式。

超时 + 手动取消组合

javascript 
async function request(url, { timeout = 10_000, signal: externalSignal } = {}) {
  const signal = AbortSignal.any([
    AbortSignal.timeout(timeout),
    ...(externalSignal ? [externalSignal] : []),
  ]);
  try {
    const res = await fetch(url, { signal });
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    return await res.json();
  } catch (err) {
    if (err.name === "TimeoutError") throw new Error("请求超时");
    if (err.name === "AbortError") throw new Error("已取消");
    throw err;
  }
}

可取消的 sleep

javascript 
function sleep(ms, { signal } = {}) {
  return new Promise((resolve, reject) => {
    if (signal?.aborted) return reject(signal.reason);
    const timer = setTimeout(resolve, ms);
    signal?.addEventListener("abort", () => {
      clearTimeout(timer);
      reject(signal.reason);
    }, { once: true });
  });
}

// Node 原生：
import { setTimeout as sleep } from "node:timers/promises";
await sleep(1000, undefined, { signal });

04 典型代码

带取消的重试

javascript 
async function retry(fn, { maxAttempts = 3, baseMs = 200, signal } = {}) {
  let lastErr;
  for (let i = 0; i < maxAttempts; i++) {
    signal?.throwIfAborted();               // 每次尝试前检查
    try { return await fn(signal); }
    catch (err) {
      if (err.name === "AbortError") throw err;  // 取消立即退出
      lastErr = err;
    }
    await sleep(baseMs * 2 ** i, { signal });    // 可取消的 sleep
  }
  throw new Error(`Failed after ${maxAttempts}`, { cause: lastErr });
}

// 用法
const controller = new AbortController();
setTimeout(() => controller.abort(), 5000);
await retry((s) => fetch(url, { signal: s }), { signal: controller.signal });

可取消的 async iterator

javascript 
async function* fetchPages(base, { signal } = {}) {
  let page = 1;
  while (true) {
    signal?.throwIfAborted();
    const res = await fetch(`${base}?p=${page}`, { signal });
    const { items, hasMore } = await res.json();
    yield* items;
    if (!hasMore) return;
    page++;
  }
}

const controller = new AbortController();
try {
  for await (const item of fetchPages("/api", { signal: controller.signal })) {
    if (shouldStop(item)) { controller.abort(); break; }
    process(item);
  }
} catch (err) {
  if (err.name !== "AbortError") throw err;
}

并发请求 + 超时 + 早退

javascript 
async function raceFastest(urls, { timeout = 5000 } = {}) {
  const controller = new AbortController();
  const signal = AbortSignal.any([controller.signal, AbortSignal.timeout(timeout)]);
  try {
    const winner = await Promise.any(urls.map(u => fetch(u, { signal })));
    controller.abort();    // 取消其他未决 fetch
    return winner;
  } catch (err) {
    if (err instanceof AggregateError) throw new Error("all mirrors failed", { cause: err });
    throw err;
  }
}

事件模拟 once + 自动 remove

javascript 
// 等待点击事件（带超时）
function waitClick(el, { timeout = 3000 } = {}) {
  return new Promise((resolve, reject) => {
    const controller = new AbortController();
    const timeoutSig = AbortSignal.timeout(timeout);
    timeoutSig.addEventListener("abort", () => {
      controller.abort();
      reject(timeoutSig.reason);
    }, { once: true });
    el.addEventListener("click", (e) => {
      controller.abort();
      resolve(e);
    }, { signal: controller.signal, once: true });
  });
}

05 常见陷阱

AbortError vs 其他错误

fetch 被 abort 时 reject 为 DOMException 且 err.name === "AbortError"。不是 err instanceof AbortError（没这个全局 class）。AbortSignal.timeout 抛的是 "TimeoutError"。业务分支要按 err.name 或 err.code 判断。

Signal 一次性，不能复用

controller abort 后 signal 永远 aborted。想要"可以取消并重启"的场景，每次新建一个 AbortController。

忘了检查 signal.aborted

原生 fetch 会自己观察 signal。但你自己写的 async 函数必须主动检查 —— 否则即使外部 abort，函数仍跑完。关键位置用 signal?.throwIfAborted()。

React Strict Mode 的双 effect 调用

React 18 dev mode 下 effect 会跑两遍 → 第一次的 controller 立刻被 cleanup abort 掉。如果你依赖"看到 AbortError 就认为是组件卸载"，开发环境会有误报。用 if (err.name !== "AbortError") 过滤掉 AbortError 再上报。

AbortSignal.any 在老浏览器 / Node 不支持

ES2024 特性，Node 20+ / Chrome 116+ / Safari 17.4+。老环境要 polyfill 或手写 "合并两个 signal" 的工具函数。

signal abort 后 listener 不再触发

aborted 是终态，已经 aborted 的 signal 不会再 fire "abort" 事件。如果你在 aborted 之后才 addEventListener("abort", fn)，fn 永远不会调用 —— 必须先检查 signal.aborted。

fetch 后再 abort 不会撤销响应

如果 fetch 已经 resolved（响应头回来了），controller.abort() 会让响应体读取抛 AbortError，但"已经下载的字节"仍在。对于"不想让响应到达"的场景，要在 fetch resolve 之前 abort。

controller.signal === controller.signal

每次访问 controller.signal 返回同一个对象（不是新建）。不用担心多次访问性能或身份问题。
