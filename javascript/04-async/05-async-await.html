async / await

async/await 不是新 API —— 它是 Generator + Promise + 自动恢复的语法糖。理解它的去糖结果，你就能精确预测执行时序、异常行为、循环中的并发特征，并不再被 "forEach + async" 之类的陷阱骗到。

阅读时长 ~20 分钟 前置 03 Promise · P1-13 iterator 对应 javascript.info Async/await

01 速览

javascript 
// 4 种形式
async function fn() { }                 // 声明
const fn = async function () { };       // 表达式
const fn = async () => { };             // 箭头
class X { async method() { } }          // class 方法

// async 函数特性
async function hi() { return 42; }
hi();                                    // 返回 Promise<42>

async function fail() { throw new Error("x"); }
fail();                                  // 返回 rejected Promise

// await 只能在 async 函数里（或 ESM 顶层）
async function loadUser() {
  const res = await fetch("/api/user");
  const data = await res.json();
  return data;
}

// 等价 Promise 链
function loadUser2() {
  return fetch("/api/user").then(r => r.json());
}

02 底层逻辑

async / await 的去糖

规范上 async function 可以用 Generator + 自动 resume 来等价表达：

javascript 
// 原始
async function fn() {
  const a = await fetchA();
  const b = await fetchB(a);
  return a + b;
}

// 等价去糖（大致）
function fn() {
  return spawn(function* () {
    const a = yield fetchA();
    const b = yield fetchB(a);
    return a + b;
  });
}

// spawn 是"自动 resume generator"的工具
function spawn(genFn) {
  return new Promise((resolve, reject) => {
    const gen = genFn();
    step(gen.next());
    function step({ value, done }) {
      if (done) return resolve(value);
      Promise.resolve(value).then(
        v => step(gen.next(v)),
        e => step(gen.throw(e)),
      );
    }
  });
}

await 的规范语义

await expr 的规范步骤：

  1. 计算 expr，把结果用 Promise.resolve 包装（thenable 也被吸收）。
  2. "暂停"当前 async 函数，返回一个 pending Promise 给调用者。
  3. 给内部 Promise 挂 .then / .catch 回调 —— 在 microtask 中 resume。
  4. resolve 时用值继续执行（作为 await 表达式的值）；reject 时在 await 处抛错。
javascript 
// 所以每个 await 都"让出"一轮 microtask
async function f() {
  console.log(1);
  await Promise.resolve();   // 让出
  console.log(2);            // 下一轮 microtask
  await Promise.resolve();
  console.log(3);            // 又让一轮
}
console.log(0);
f();
console.log(4);
// 输出：0, 1, 4, 2, 3

Top-level await（ESM）

ES2022 起 ESM 顶层可以直接 await（非 ESM 脚本/CJS 不行）：

javascript 
// module.mjs
const config = await fetch("/config.json").then(r => r.json());
export { config };

// 引入此模块的代码也会"异步"等待：
// import { config } from "./module.mjs";   // 等 config 加载完才能用

// ⚠ 影响模块加载时序：所有 depend on 此模块的都被阻塞
// 生产代码里谨慎使用 —— 启动期初始化可以；热路径别用

await vs .then 的等价关系

写法 等价 .then
const v = await p; p.then(v => { /* 后续代码 */ })
try { await p; } catch (e) { } p.then(_ => {}, e => {})
await Promise.all([a, b]) Promise.all([a, b]).then(...)
for (const p of ps) await p; 顺序：每个 .then 链起来
await Promise.all(ps.map(fn)) 并行：所有 fn 同时启动

错误处理

async 函数里的 throw → 返回 rejected Promise。
await rejectedPromise → 在 await 处"同步抛错"，被 try/catch 捕获。

javascript 
async function f() {
  try {
    await fetch("/bad");      // reject 了
  } catch (err) {
    console.error("caught:", err);
    throw new Error("wrap", { cause: err });   // 被调用方接
  } finally {
    cleanup();
  }
}

// 上层
try {
  await f();
} catch (err) {
  err.message;    // "wrap"
  err.cause;      // 原始错误
}

循环中的并发性

javascript 
// ❌ 顺序 —— 每个请求等上一个完成（1 + 1 + 1 = 3s）
for (const url of urls) {
  await fetch(url);
}

// ✓ 并行 —— 所有请求同时启动（max = 1s）
await Promise.all(urls.map(u => fetch(u)));

// ✓ 并发但需要顺序结果：allSettled + 处理
const results = await Promise.all(urls.map(u => fetch(u).then(r => r.json())));
// results 顺序和 urls 一致

// 顺序但要求"前一个结果用于下一个"：必须 for 循环 + await
let acc = initial;
for (const item of items) {
  acc = await process(item, acc);    // 每步依赖前一步
}

03 应用场景

重构 Promise 链到 async/await

javascript 
// 链式
function loadDashboard() {
  return fetchUser()
    .then(user => Promise.all([user, fetchOrders(user.id), fetchStats(user.id)]))
    .then(([user, orders, stats]) => ({ user, orders, stats }))
    .catch(err => { logger.error(err); return null; });
}

// async/await
async function loadDashboard() {
  try {
    const user = await fetchUser();
    const [orders, stats] = await Promise.all([
      fetchOrders(user.id),
      fetchStats(user.id),
    ]);
    return { user, orders, stats };
  } catch (err) {
    logger.error(err);
    return null;
  }
}

顶层 await 做模块初始化

javascript 
// db.mjs
import { Pool } from "pg";
const pool = new Pool({ connectionString: process.env.DATABASE_URL });
await pool.connect();           // 启动时确保数据库可用
export { pool };

// 使用方
import { pool } from "./db.mjs";  // import 本身完成时 pool 已就绪
pool.query("...");

⚠ 生产代码慎用 —— 连锁依赖的模块都会被阻塞，影响启动时间。

Fire-and-forget 模式

javascript 
// 不需要等待结果的副作用
function trackEvent(event) {
  sendBeacon(event).catch(err => logger.warn({ err }, "tracking failed"));
  //                ↑ 必须 .catch 防止 unhandled rejection
}

// 或用 void 显式标记"故意不 await"
void sendBeacon(event);    // ⚠ 抛错仍会变 unhandled
// 更好：
sendBeacon(event).catch(noop);

异步中间件链（Koa / fastify 风格）

javascript 
async function compose(middlewares) {
  return async (ctx) => {
    let index = -1;
    async function dispatch(i) {
      if (i <= index) throw new Error("next called multiple times");
      index = i;
      const mw = middlewares[i];
      if (mw) await mw(ctx, () => dispatch(i + 1));
    }
    await dispatch(0);
  };
}

// 中间件可以 await 下游
app.use(async (ctx, next) => {
  const start = Date.now();
  await next();                   // 等下游全部完成
  ctx.responseTime = Date.now() - start;
});

async generator + for await of

javascript 
// 分页流
async function* fetchPages(base) {
  let page = 1;
  while (true) {
    const { items, hasMore } = await fetch(`${base}?p=${page}`).then(r => r.json());
    yield* items;                    // 逐项产出
    if (!hasMore) return;
    page++;
  }
}

for await (const item of fetchPages("/api")) {
  process(item);
  if (shouldStop(item)) break;       // 提前退出自动 close generator
}

04 典型代码

forEach + async 的坑

javascript 
// ❌ forEach 不等 await
await [1, 2, 3].forEach(async (x) => {
  await save(x);             // save 同时启动，但 forEach 立即返回
});
// forEach 返回 undefined，await undefined 不等任何事
// 结果：save(1/2/3) 并行，外层代码不等它们

// ✓ 并行：Promise.all + map
await Promise.all([1, 2, 3].map(async (x) => save(x)));

// ✓ 串行：for-of
for (const x of [1, 2, 3]) await save(x);

// ✓ 并发限制：p-limit / Promise.all 加信号量（下一章 P3-07）

ESM 顶层 await

javascript 
// 动态 import 替换
// 旧：
let config;
(async () => { config = await fetch(...).then(r => r.json()); })();

// ES2022 顶层 await
const config = await fetch("/api/config").then(r => r.json());

// 条件加载
const mod = process.env.NODE_ENV === "development"
  ? await import("./dev-tools.mjs")
  : null;

async IIFE（非 ESM 场景）

javascript 
// 旧代码（CJS / script）没有顶层 await，用 IIFE 包
(async () => {
  const data = await load();
  initialize(data);
})().catch(err => {
  console.error(err);
  process.exit(1);
});

精确时序

javascript 
async function test() {
  console.log("a");
  await null;              // await 任何值都让出一轮 microtask
  console.log("b");
  await null;
  console.log("c");
}

console.log("1");
test();
console.log("2");
Promise.resolve().then(() => console.log("p1"));

// 输出：1, a, 2, b, p1, c
// 解释：
// - "1": 同步
// - test() 启动 → "a" 同步 → await 让出
// - "2": 同步
// - microtask：test 恢复 → "b" → await 让出
// - microtask：p1 队列里（先入先出）先执行 → "p1"
// - 下一轮 microtask：test 恢复 → "c"

05 常见陷阱

forEach(async fn) 不等待

forEach 签名是 sync，返回 undefined。传 async 函数进去，回调返回的 Promise 被丢弃。用 for-of 或 Promise.all(arr.map(async fn))。ESLint no-misleading-character-class / 社区规则能拦。

循环里 await 是串行

for (const url of urls) await fetch(url); 是 1 + 1 + 1 + ... 的串行。100 个 url × 100ms = 10s。不需要顺序就用 Promise.all(urls.map(fetch))。

忘 await 导致错误逃逸

async function f() { doAsync(); } —— 没 await 的 doAsync() 里的错误不会进 f 的 reject 链，变 unhandled rejection。ESLint @typescript-eslint/no-floating-promises / promise/always-return 能拦。

return await p 还是 return p？

两者行为几乎一样（async 函数自动 unwrap Promise），但有差异：错误堆栈和 try/catch 位置。try { return await p } catch 能接住 p 的 reject；try { return p } catch 接不住（p 的 reject 发生在 try 块外）。规范建议"总是 return await"便于错误处理。但 V8 优化让两者性能等同。

async 函数里 throw 不是同步

async function f() { throw x; } f(); 返回 rejected Promise，不是同步抛错。调用 f 的人必须 try/await 或 .catch 才能接。

await 0 / await "x" 也让出一轮 microtask

即使表达式不是 Promise，await 仍走 Promise.resolve(val).then —— 至少让出一次。代码里看到 await null 作为"让行" trick，就是利用这个。

Top-level await 阻塞所有依赖模块

如果 a.mjs 顶层 await 5s，所有 import a 的模块都要等 5s 才能开始执行。深度依赖链会让启动时间炸掉。仅在"真正需要异步初始化"的叶子模块用。

async 不能用在构造器

class Foo { async constructor() {} } 语法错。构造器必须同步返回实例。需要异步初始化用工厂方法：static async create() { const foo = new Foo(); await foo.#init(); return foo; }。

await 在非 async 函数里是 SyntaxError

function f() { await p; } 报错。除非是 ESM 顶层。移动 await 到 async 函数，或用 p.then(...)。
