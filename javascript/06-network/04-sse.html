Server-Sent Events

SSE 是"HTTP 上的长连接单向推送"：服务器保持连接打开，持续往 response body 追加 text/event-stream 格式的消息。比 WebSocket 简单（走 HTTP）、自带重连、带事件类型。缺点是单向（server → client）和浏览器不支持自定义 header —— 这让 fetch + stream 模式常被用作"升级版 SSE"。

阅读时长 ~18 分钟 前置 01 Fetch · 03 WebSocket MDN Server-sent events

01 速览

javascript 
// EventSource API
const es = new EventSource("/api/stream", {
  withCredentials: false,   // 是否带 cookie（对同源默认带）
});

// 默认事件
es.addEventListener("message", (e) => {
  console.log(e.data);       // 字符串
  console.log(e.lastEventId);
});

// 自定义事件（服务端 event: xxx）
es.addEventListener("price-update", (e) => {
  const tick = JSON.parse(e.data);
});

// 错误 + 自动重连
es.addEventListener("error", () => {
  console.log("readyState", es.readyState);
  // 0 CONNECTING（会自动重连）
  // 1 OPEN
  // 2 CLOSED（永久关闭，不会重连）
});

// 关闭
es.close();

// readyState 常量（同 WebSocket）
EventSource.CONNECTING;  // 0
EventSource.OPEN;        // 1
EventSource.CLOSED;      // 2

text/event-stream 格式

text 
Content-Type: text/event-stream
Cache-Control: no-cache
Connection: keep-alive

data: hello

data: {"price": 100}

event: price-update
data: {"symbol": "BTC", "price": 50000}
id: 42
retry: 5000

: 这是注释（以冒号开头）—— 用于心跳
  • data: 消息体（可多行，合并时用 \n）
  • event: 自定义事件名（不写默认 "message"）
  • id: 事件 ID（客户端记住，重连时发 Last-Event-Id）
  • retry: 客户端重连延迟（ms）
  • 空行作消息结束标志

02 底层逻辑

自动重连机制

EventSource 的最大便利：自动重连。默认延迟 3 秒（可被 retry: 字段覆盖）。每次重连时，客户端把最后收到的 id 放入 Last-Event-Id 请求 header，服务端可据此断点续传：

text 
第一次连接：
GET /stream HTTP/1.1
Accept: text/event-stream

服务响应：
data: msg1
id: 1

data: msg2
id: 2

（连接意外中断）

客户端自动重连：
GET /stream HTTP/1.1
Accept: text/event-stream
Last-Event-Id: 2      ← 自动带上

服务端从 id > 2 的消息继续推送

SSE vs WebSocket

SSE WebSocket
方向 单向（server → client） 全双工
协议 HTTP / HTTPS ws:// wss://
格式 文本（UTF-8） 文本 + 二进制
重连 内置 + Last-Event-Id 需自己实现
事件类型 原生支持 手动 JSON 分类
自定义 header ✗（浏览器 EventSource） ✗
走 HTTP 中间设施 ✓（代理、CDN 熟悉） ✓（但需要 Upgrade 支持）
HTTP/2 复用 ✓（多路复用，多个连接共享） ✗
浏览器数量限制 HTTP/1.1 下 6 个/域 类似

选择：

  • SSE：通知、feed、AI token 流、股价推送等只下行的场景
  • WebSocket：聊天、协同编辑、游戏等需要双向低延迟
  • fetch stream：需要 POST / 自定义 header 的流式响应（最灵活）

HTTP/2 下的复用红利

HTTP/1.1 下浏览器对同一域名最多 6 个并发连接。SSE 长连占一个 → 剩 5 个给普通请求。同时开多个 SSE 就卡死。HTTP/2 多路复用一个 TCP 连接可跑多个流，SSE 不再占用连接预算 —— 这是 HTTP/2+ 时代 SSE 被重新推崇的原因。

fetch stream 替代 EventSource

浏览器 EventSource 有两个硬限制：只能 GET、不能加自定义 header。想发 Authorization / 想 POST 请求体 → 用 fetch 手动解析 SSE：

javascript 
async function sseLike(url, { body, headers, signal } = {}) {
  const res = await fetch(url, {
    method: body ? "POST" : "GET",
    headers: {
      "Accept": "text/event-stream",
      "Cache-Control": "no-cache",
      ...headers,
      ...(body ? { "Content-Type": "application/json" } : {}),
    },
    body: body ? JSON.stringify(body) : undefined,
    signal,
  });
  const reader = res.body.getReader();
  const decoder = new TextDecoder();
  let buffer = "";

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    buffer += decoder.decode(value, { stream: true });
    const events = buffer.split("\n\n");
    buffer = events.pop();         // 最后一个可能不完整
    for (const raw of events) {
      const fields = Object.fromEntries(raw.split("\n").map(l => {
        const i = l.indexOf(":");
        return [l.slice(0, i), l.slice(i + 1).trimStart()];
      }));
      if (fields.data !== undefined) yield fields;
    }
  }
}

// 用法
for await (const msg of sseLike("/api/chat", {
  body: { prompt },
  headers: { Authorization: `Bearer ${token}` },
})) {
  appendToken(msg.data);
}

实时通知 / feed

javascript 
const es = new EventSource("/api/notifications");

es.addEventListener("notification", (e) => {
  const notif = JSON.parse(e.data);
  showToast(notif);
  incrementBadge();
});

es.addEventListener("error", () => {
  // 自动重连（不需要手动）—— EventSource 默认会
  // 如果 es.readyState === 2（CLOSED），服务端主动关闭了 → 不再重连

AI token 流（ChatGPT 风格）

javascript 
// OpenAI / Anthropic / 自建 LLM API 都用 SSE 流式返回
// 但需要 POST 和 Authorization → 用 fetch stream

async function* chat(prompt) {
  const res = await fetch("/api/chat/completions", {
    method: "POST",
    headers: {
      "Accept": "text/event-stream",
      "Authorization": `Bearer ${token}`,
      "Content-Type": "application/json",
    },
    body: JSON.stringify({ prompt, stream: true }),
  });
  const reader = res.body.getReader();
  const decoder = new TextDecoder();
  let buffer = "";

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    buffer += decoder.decode(value, { stream: true });
    const lines = buffer.split("\n");
    buffer = lines.pop();
    for (const line of lines) {
      if (!line.startsWith("data: ")) continue;
      const data = line.slice(6);
      if (data === "[DONE]") return;
      yield JSON.parse(data).choices[0].delta.content;
    }
  }
}

for await (const token of chat("Hello")) process.stdout.write(token);

多 tab 共享 SSE（避免重复连接）

// shared-sse.worker.js
const es = new EventSource("/api/stream");
const ports = new Set();
onconnect = (e) => {
  const port = e.ports[0];
  ports.add(port);
  port.start();
};
es.onmessage = (e) => ports.forEach(p => p.postMessage(e.data));

04 典型代码

完整 SSE 解析器（for fetch stream）

javascript 
async function* parseSSE(stream) {
  const reader = stream.getReader();
  const decoder = new TextDecoder();
  let buffer = "";
  let event = { data: "", event: "message", id: "", retry: 0 };

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    buffer += decoder.decode(value, { stream: true });

    let lineEnd;
    while ((lineEnd = buffer.indexOf("\n")) >= 0) {
      const line = buffer.slice(0, lineEnd).replace(/\r$/, "");
      buffer = buffer.slice(lineEnd + 1);

      if (line === "") {
        if (event.data) yield event;
        event = { data: "", event: "message", id: "", retry: 0 };
      } else if (line.startsWith(":")) {
        // comment / heartbeat, skip
      } else {
        const sep = line.indexOf(":");
        const key = sep > 0 ? line.slice(0, sep) : line;
        let value = sep > 0 ? line.slice(sep + 1) : "";
        if (value.startsWith(" ")) value = value.slice(1);
        if (key === "data") event.data = event.data ? event.data + "\n" + value : value;
        else if (key === "event") event.event = value;
        else if (key === "id") event.id = value;
        else if (key === "retry") event.retry = Number(value);
      }
    }
  }
  if (event.data) yield event;
}

fetch SSE 带重连

javascript 
async function* robustSSE(url, options = {}) {
  let lastEventId = "";
  let retryMs = 3000;

  while (true) {
    try {
      const res = await fetch(url, {
        ...options,
        headers: {
          "Accept": "text/event-stream",
          ...(lastEventId ? { "Last-Event-Id": lastEventId } : {}),
          ...options.headers,
        },
      });
      if (!res.ok) throw new Error(`HTTP ${res.status}`);

      for await (const event of parseSSE(res.body)) {
        if (event.id) lastEventId = event.id;
        if (event.retry) retryMs = event.retry;
        yield event;
      }
    } catch (err) {
      if (options.signal?.aborted) throw err;
      console.warn("SSE error, reconnecting in", retryMs, "ms", err);
      await new Promise(r => setTimeout(r, retryMs));
    }
  }
}

原生 EventSource 的 wrapper

javascript 
function subscribe(url, handlers) {
  const es = new EventSource(url);
  for (const [event, fn] of Object.entries(handlers)) {
    es.addEventListener(event, fn);
  }
  return es;   // 调用方可以 .close()
}

const es = subscribe("/stream", {
  "new-order": (e) => handleOrder(JSON.parse(e.data)),
  "user-update": (e) => handleUpdate(JSON.parse(e.data)),
  error: () => console.warn("connection issue"),
});

app.get("/stream", (req, res) => {

05 常见陷阱

EventSource 不支持自定义 header

只能走 cookie 或 URL 参数鉴权。想自定义 header 一定要用 fetch stream（见本章示例）或用第三方库（@microsoft/fetch-event-source 等）。

EventSource 只能 GET

没法 POST 请求体。要发数据参数只能 URL 查询字符串（限长度、泄漏日志）或先 POST 创建会话再 SSE 订阅。

Nginx / 代理缓冲

Nginx 默认缓冲响应 → SSE 消息积累到 buffer 满才转发 → 客户端看不到实时。服务端要设 X-Accel-Buffering: no header 或 Nginx 配置 proxy_buffering off。

浏览器 6 连接限制（HTTP/1.1）

同域名最多 6 个 HTTP/1.1 连接。SSE 长连占一个 → 剩 5 个给普通请求。多 tab 各开一个 SSE 就会把连接池耗光。HTTP/2+ 多路复用解决这个。或用 SharedWorker 共享连接。

重连不总是触发

es.readyState === 2（CLOSED）时不会自动重连（服务端主动关闭、或客户端 es.close()）。只有连接异常断开（readyState 回到 0 CONNECTING）才自动重连。

Last-Event-Id 需要服务端支持

浏览器自动发，但服务端不写 id: 字段就没意义。用 SSE 做可靠推送必须服务端维护消息顺序 + id。

SSE 只是 UTF-8 文本

不能传 ArrayBuffer / Blob。二进制数据要 base64 编码（体积增加 33%）或用 WebSocket。

服务端忘记心跳导致代理断连

Cloudflare / ALB / Nginx 等 60s 空闲就断。SSE 的注释行（: heartbeat\n\n）可以作心跳。服务端每 30s 发一个保活。
