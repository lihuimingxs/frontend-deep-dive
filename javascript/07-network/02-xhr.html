XHR 残余场景

XMLHttpRequest 是 fetch 之前唯一的 AJAX API。今天仍有三个场景 fetch 覆盖不好：上传进度（fetch 的 upload stream API 较新）、同步请求（极少用但仍必要）、带 XHR 注入的老库。本章把 XHR 最少够用的知识点 + 何时选 fetch 讲清楚。

阅读时长 ~15 分钟 前置 01 Fetch MDN XMLHttpRequest

01 速览

javascript 
// 基本 API
const xhr = new XMLHttpRequest();
xhr.open("GET", "/api/data", true);       // true = async（false 是同步）
xhr.setRequestHeader("X-Token", "abc");
xhr.responseType = "json";                // "text" / "json" / "blob" / "arraybuffer" / "document"
xhr.timeout = 10_000;

xhr.onload = () => console.log(xhr.response);
xhr.onerror = () => console.error("network error");
xhr.ontimeout = () => console.error("timeout");
xhr.upload.onprogress = (e) => {
  if (e.lengthComputable) console.log(e.loaded / e.total);
};
xhr.send(JSON.stringify(data));

// 取消
xhr.abort();

XHR vs fetch 对比

XHR fetch
API 风格 事件 + 属性 Promise
上传进度 ✓（xhr.upload.onprogress） ✓（Chrome 105+，duplex: "half"）
下载进度 ✓ ✓（ReadableStream）
流式响应 △（responseType="moz-chunked-arraybuffer" 历史 hack） ✓（res.body）
取消 xhr.abort() AbortController
默认 credentials same-origin same-origin
同步请求 ✓（不推荐） ✗
CORS 精细控制 有限 更丰富

02 底层逻辑

readyState 状态机

值 状态 含义
0 UNSENT new 后未调 open
1 OPENED 调过 open，未 send
2 HEADERS_RECEIVED 响应头已到
3 LOADING response body 下载中
4 DONE 完成（成功或失败）

XHR 事件全景

javascript 
// 下载
xhr.onloadstart = () => {};       // 开始
xhr.onprogress = (e) => {         // 下载进度
  if (e.lengthComputable) progress = e.loaded / e.total;
};
xhr.onload = () => {};             // 下载完成（含 HTTP 错误）
xhr.onerror = () => {};            // 网络错误
xhr.ontimeout = () => {};          // 超时
xhr.onabort = () => {};            // 被 abort
xhr.onloadend = () => {};          // 总是最后触发

// 上传
xhr.upload.onloadstart = () => {};
xhr.upload.onprogress = (e) => {};  // ← 最重要的
xhr.upload.onload = () => {};
xhr.upload.onerror = () => {};
xhr.upload.ontimeout = () => {};
xhr.upload.onabort = () => {};
xhr.upload.onloadend = () => {};

// 老式：onreadystatechange
xhr.onreadystatechange = () => {
  if (xhr.readyState === 4) { /* DONE */ }
};

responseType 和 response

javascript 
xhr.responseType = "json";        // xhr.response 自动 JSON.parse
xhr.responseType = "blob";        // 拿到 Blob（适合文件下载）
xhr.responseType = "arraybuffer"; // 二进制数据
xhr.responseType = "document";    // 解析 XML/HTML
xhr.responseType = "text";        // 默认
xhr.responseType = "";            // 同 "text"

// ⚠ 必须在 send() 前设
// 设后 xhr.responseText 和 xhr.responseXML 对非 text 类型会抛错
// 用 xhr.response 统一访问

上传进度 —— XHR 仍最简单

javascript 
function upload(file, onProgress) {
  return new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest();
    xhr.upload.addEventListener("progress", (e) => {
      if (e.lengthComputable) onProgress(e.loaded / e.total);
    });
    xhr.addEventListener("load", () => {
      if (xhr.status >= 200 && xhr.status < 300) resolve(xhr.response);
      else reject(new Error(`HTTP ${xhr.status}`));
    });
    xhr.addEventListener("error", () => reject(new Error("network error")));
    xhr.addEventListener("abort", () => reject(new DOMException("aborted", "AbortError")));
    xhr.responseType = "json";
    xhr.open("POST", "/upload");
    xhr.send(file);
  });
}

// fetch 的等价方案要 ReadableStream + duplex: "half"，支持仅限 Chrome 105+
// XHR 是全浏览器、全 Node 都支持的

同步 XHR（极少场景）

javascript 
const xhr = new XMLHttpRequest();
xhr.open("GET", url, false);       // false = sync
xhr.send();
// 整个主线程阻塞直到返回 —— 浏览器会警告
console.log(xhr.responseText);

// 唯一合理场景：unload 时上报（被 sendBeacon / fetch keepalive 替代）
// 老代码可能仍有：XHR 同步加载小配置（启动期）

// 规范 deprecation：
// - 主线程同步 XHR 已经警告多年
// - Chrome 可能未来禁用
// - Worker 里同步 XHR 仍允许（Worker 本来就阻塞主线程安全）

用 AbortSignal 给 XHR 加现代取消（polyfill 思路）

javascript 
function xhrWithSignal(options, signal) {
  const xhr = new XMLHttpRequest();
  if (signal?.aborted) { xhr.abort(); throw signal.reason; }
  signal?.addEventListener("abort", () => xhr.abort(), { once: true });
  // ... 其余逻辑
  return xhr;
}

// axios 1.x 已原生支持 signal 参数，底层是 XHR + 监听 abort 事件

分片上传 + 并发

javascript 
// 大文件切片，并发上传，每片带进度
async function uploadChunks(file, { chunkSize = 5 * 1024 * 1024, concurrency = 3 }) {
  const chunks = [];
  for (let i = 0; i < file.size; i += chunkSize) {
    chunks.push(file.slice(i, i + chunkSize));
  }

  const progress = new Array(chunks.length).fill(0);

  async function uploadOne(chunk, index) {
    return new Promise((resolve, reject) => {
      const xhr = new XMLHttpRequest();
      xhr.upload.onprogress = (e) => {
        progress[index] = e.loaded;
        updateTotal();
      };
      xhr.onload = () => resolve();
      xhr.onerror = () => reject(new Error(`chunk ${index} failed`));
      xhr.open("POST", `/upload/chunk?index=${index}`);
      xhr.send(chunk);
    });
  }

  // 限并发（见 P3-07）
  await pLimit(concurrency)(chunks.map((chunk, i) => () => uploadOne(chunk, i)));
  await fetch(`/upload/complete?total=${chunks.length}`, { method: "POST" });
}

兼容 XHR 代码的迁移

javascript 
// Promise 包装器 —— 从老代码迁移时保留 XHR 特性
function request(method, url, { body, headers = {}, timeout, responseType, onUploadProgress, signal } = {}) {
  return new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest();
    xhr.open(method, url, true);
    for (const [k, v] of Object.entries(headers)) xhr.setRequestHeader(k, v);
    if (responseType) xhr.responseType = responseType;
    if (timeout) xhr.timeout = timeout;
    if (onUploadProgress) xhr.upload.addEventListener("progress", onUploadProgress);

    xhr.onload = () => {
      if (xhr.status >= 200 && xhr.status < 300) resolve(xhr.response);
      else reject(new HttpError(xhr.status, xhr.response, url));
    };
    xhr.onerror = () => reject(new Error("network"));
    xhr.ontimeout = () => reject(new Error("timeout"));
    xhr.onabort = () => reject(new DOMException("aborted", "AbortError"));

    if (signal) {
      if (signal.aborted) return reject(signal.reason);
      signal.addEventListener("abort", () => xhr.abort(), { once: true });
    }

    xhr.send(body);
  });
}

04 典型代码

上传 UI + 断点续传

javascript 
class FileUploader extends EventTarget {
  constructor() {
    super();
    this.xhr = null;
  }
  upload(file, url) {
    const xhr = this.xhr = new XMLHttpRequest();
    xhr.upload.addEventListener("progress", (e) => {
      if (e.lengthComputable) {
        this.dispatchEvent(new CustomEvent("progress", {
          detail: { loaded: e.loaded, total: e.total, ratio: e.loaded / e.total },
        }));
      }
    });
    xhr.addEventListener("load", () => {
      this.dispatchEvent(new CustomEvent("complete", { detail: xhr.response }));
    });
    xhr.addEventListener("error", () => this.dispatchEvent(new Event("error")));
    xhr.responseType = "json";
    xhr.open("POST", url);
    xhr.send(file);
  }
  pause() { this.xhr?.abort(); }   // 实际"断点续传"要配合后端 range
}

fetch 的上传进度（新方案）

javascript 
// Chrome 105+ / Node 18+
async function fetchUpload(url, file, onProgress) {
  let loaded = 0;
  const stream = new ReadableStream({
    async start(controller) {
      const reader = file.stream().getReader();
      while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        loaded += value.length;
        onProgress(loaded / file.size);
        controller.enqueue(value);
      }
      controller.close();
    },
  });
  return fetch(url, {
    method: "POST",
    body: stream,
    duplex: "half",
    headers: { "Content-Type": "application/octet-stream" },
  });
}
// 兼容性差：Safari 尚未支持。老项目仍用 XHR 更稳

05 常见陷阱

主线程同步 XHR 已弃用

xhr.open(..., false) 在浏览器 devtools 有警告，Chrome 宣布未来禁用。几乎所有老代码都应迁移到 fetch + keepalive 或 sendBeacon。

忘记在 open 前设置 responseType

xhr.responseType = "json" 必须在 open() 之后、send() 之前调用。调用顺序错会抛 InvalidStateError。

4xx/5xx 不触发 onerror

和 fetch 一样，xhr.onerror 只在网络失败（断网、CORS 拒绝、DNS 失败）时触发。HTTP 错误走 onload，要检查 xhr.status。

Chunked response 的 responseText 增长

流式响应期间 xhr.responseText 包含已下载的所有字节 —— 每次 progress 都解析整个字符串会 O(n²)。用指针记录 lastIndex 只处理新增。

跨域同步 XHR 不能设某些 header

CORS 限制 + 同步 XHR 组合时某些行为更严格。现代代码不要依赖这个路径。

timeout 只对异步 XHR 生效

同步 XHR 的 xhr.timeout 会抛 InvalidAccessError（规范禁止）。

事件 vs 属性赋值

xhr.onload = fn 和 xhr.addEventListener("load", fn) 都支持，但属性赋值只能挂一个 —— 多次赋值后者覆盖前者。工程中推荐 addEventListener。

XHR 的 progress 事件可能没有 lengthComputable

服务器没发 Content-Length → e.lengthComputable 是 false，e.total 是 0。进度条要显示"已下载 X MB"而不是百分比。
