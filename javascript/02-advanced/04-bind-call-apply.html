bind / call / apply

三个方法都住在 Function.prototype 上，都用于显式绑定 this。区别在 "立即调用 vs 返回新函数"、"参数列表 vs 参数数组"。本章给出判定表、手写 polyfill（帮你精确理解规范），以及箭头函数和 class field 下的边界情况。

阅读时长 ~15 分钟 前置 03 this 四规则 对应 javascript.info Decorators and forwarding, call/apply

01 速览

方法 签名 行为
fn.call(thisArg, a, b, c) this + 参数列表 立即以 thisArg 为 this 调用
fn.apply(thisArg, [a, b, c]) this + 参数数组 立即调用（同上）
fn.bind(thisArg, a, b) this + 预设参数 返回新函数，调用时 this 永固定
Reflect.apply(fn, thisArg, args) 明确三要素 推荐替代 fn.apply
javascript 
function greet(greeting, punctuation) {
  return `${greeting}, ${this.name}${punctuation}`;
}

const user = { name: "Alice" };

greet.call(user, "Hi", "!");      // "Hi, Alice!"  立即调用
greet.apply(user, ["Hi", "!"]);   // "Hi, Alice!"  立即调用，参数数组
const bound = greet.bind(user, "Hi");    // 返回新函数；this 和首参已冻结
bound("!");                       // "Hi, Alice!"  之后补余下参数

02 底层逻辑

call 与 apply 的规范流程

  1. 如果 fn 不是可调用对象 → TypeError。
  2. 按规则设置执行上下文的 this（严格模式直接用 thisArg；非严格会把 null/undefined 变 globalThis，原始值装箱）。
  3. 以参数列表 / 参数数组调用 fn。
  4. 返回 fn 的返回值。

bind 返回 bound function exotic object

bind 不是普通闭包，而是规范定义的特殊对象，有 3 个内部槽：

  • [[BoundTargetFunction]] —— 原始函数
  • [[BoundThis]] —— 预设 this
  • [[BoundArguments]] —— 预设参数数组

被调用时：

  • 普通调用 → 以 [[BoundThis]] 为 this，[[BoundArguments]] + 新参数一起传给 target。
  • new 调用 → 忽略 [[BoundThis]]，以新对象为 this，参数仍拼接。
javascript 
function User(name) { this.name = name; }
const BoundUser = User.bind({ x: 1 }, "Alice");

BoundUser();                    // undefined（返回 undefined）；this 是 {x:1}, name = "Alice"
new BoundUser();                // { name: "Alice" }  ← bind 的 this 被 new 忽略

BoundUser.prototype;            // undefined ← bound function 无 prototype
new BoundUser() instanceof User;// true       ← 但 instanceof 正确
                                // （因为 new 时用的还是 User.prototype）

多次 bind 无效

f1 = fn.bind(A); f2 = f1.bind(B); f2() 时 this 仍是 A。规范原因：bound function 调用时以自己的 [[BoundThis]] 为 this —— 所以二次 bind 时 [[BoundThis]] 确实被设为 B，但内部调用 target（是 f1）时又以 f1 的 [[BoundThis]]（A）覆盖。

设计哲学：bind 是单向冻结。需要不同 this 就从原始函数重新 bind。

箭头函数不受 call/apply/bind 影响

javascript 
const arrow = () => this;
arrow.call({ x: 1 });           // 外层 this
arrow.bind({ x: 1 })();         // 外层 this

// 因为箭头函数的 [[ThisMode]] 是 "lexical"，不会设置自己的 this
// call/apply/bind 试图设置的 this 被无视

Reflect.apply 的优势

fn.apply(thisArg, args) 有个历史问题：如果 fn 自身有个叫 apply 的属性被覆盖（罕见但可能），就乱套了。Reflect.apply(fn, thisArg, args) 直接调用规范算法，不依赖 fn.apply 属性：

javascript 
// 恶意对象可能遮蔽 apply
const evil = function () {};
evil.apply = () => "hacked";
evil.apply(null, []);           // "hacked"
Reflect.apply(evil, null, []);  // 仍按 Function.prototype.apply 调用

03 应用场景

借用方法：把 Array 方法用在 Array-like 上

javascript 
// 把 arguments 转数组（ES6 前的标准做法）
function fn() {
  const args = Array.prototype.slice.call(arguments);
  return args.map(x => x * 2);
}
// 现在应该用 rest 参数：function fn(...args) { ... }

// 精确类型检查
Object.prototype.toString.call(new Date());   // "[object Date]"
Object.prototype.toString.call([]);           // "[object Array]"
Object.prototype.toString.call(new Map());    // "[object Map]"

partial application（部分应用）

javascript 
function request(method, url, body) { /* ... */ }

// 用 bind 预设方法
const get  = request.bind(null, "GET");
const post = request.bind(null, "POST");
const put  = request.bind(null, "PUT");

get("/api/users");               // request("GET", "/api/users")
post("/api/users", { name });

// 如果不用 this，null 作为 thisArg；现代替代用箭头函数
const get2 = (url, body) => request("GET", url, body);
// 闭包版本避免了 bound function 的特殊性，且性能相似

事件处理器稳定 this（传统 class 组件）

javascript 
class Toggle extends Component {
  constructor(props) {
    super(props);
    this.handleClick = this.handleClick.bind(this);   // 构造器里一次 bind
  }
  handleClick() { /* this 安全 */ }
  render() {
    return <button onClick={this.handleClick}>...</button>;
    // ❌ onClick={this.handleClick.bind(this)}  每次渲染创建新函数 → 子组件不必要重渲染
  }
}

最大/最小值的 apply 技巧（已过时）

javascript 
// ES6 前
Math.max.apply(null, [1, 2, 3]);       // 3

// ES6+ 用 spread，语义更清
Math.max(...[1, 2, 3]);                // 3

// ⚠ 两者都有"参数过多（> ~64k）栈溢出"的引擎限制
// 大数组求最值用 reduce
arr.reduce((m, v) => v > m ? v : m, -Infinity);

04 典型代码

手写 myCall / myApply

javascript 
Function.prototype.myCall = function (thisArg, ...args) {
  thisArg = thisArg ?? globalThis;
  // 利用"对象方法调用的 this 是对象"这一规则
  const key = Symbol("fn");
  thisArg[key] = this;
  const result = thisArg[key](...args);
  delete thisArg[key];
  return result;
};

Function.prototype.myApply = function (thisArg, args = []) {
  return this.myCall(thisArg, ...args);
};

function greet(g) { return `${g}, ${this.name}`; }
greet.myCall({ name: "A" }, "Hi");         // "Hi, A"

手写 myBind（考虑 new 场景）

javascript 
Function.prototype.myBind = function (thisArg, ...preset) {
  const target = this;
  function BoundFn(...later) {
    // new 调用时：new.target 不为空 → this 不应被 thisArg 覆盖
    const actualThis = new.target ? this : thisArg;
    return target.apply(actualThis, [...preset, ...later]);
  }
  // 保持原型链，让 new BoundFn() instanceof target 成立
  if (target.prototype) {
    BoundFn.prototype = Object.create(target.prototype);
  }
  return BoundFn;
};

function User(name) { this.name = name; }
const F = User.myBind(null, "Alice");
new F() instanceof User;                    // true
new F().name;                               // "Alice"

基于 bind 的柯里化

javascript 
function curry(fn) {
  return function curried(...args) {
    return args.length >= fn.length
      ? fn(...args)
      : curried.bind(null, ...args);
  };
}

const add = (a, b, c) => a + b + c;
const addCurried = curry(add);
addCurried(1)(2)(3);          // 6
addCurried(1, 2)(3);          // 6
addCurried(1)(2, 3);          // 6
addCurried(1, 2, 3);          // 6

带参数的 decorator

javascript 
function withRetry(fn, maxAttempts = 3) {
  return async function (...args) {
    let lastErr;
    for (let i = 0; i < maxAttempts; i++) {
      try {
        return await fn.apply(this, args);    // ← 用 apply 保留 this
      } catch (err) {
        lastErr = err;
      }
    }
    throw lastErr;
  };
}

class API {
  async fetchUser(id) { /* ... */ }
}
API.prototype.fetchUser = withRetry(API.prototype.fetchUser);

05 常见陷阱

bind 只固化一次 this

fn.bind(A).bind(B)() 的 this 是 A，不是 B。设计如此。真的需要改，用"解包"：存原函数引用 + 重新 bind。

bind 不能改变箭头函数的 this

箭头函数无 [[ThisMode]] = "strict"/"global"，call/apply/bind 的 thisArg 传进去不生效。箭头函数永远用定义时的词法 this。

apply 参数数组过长触发 RangeError

老引擎 Math.max.apply(null, bigArr) 在几万元素时会抛 "too many arguments"。ES2015 修复了规范，现代引擎单调用可接收 ~65536+ 个参数。但仍不推荐。大数组用 reduce。

每次 bind 都返回新函数

onClick={this.fn.bind(this)} 在 render 里写，每次 render 都是新引用 → 子组件浅比较不过 → 不必要重渲染。React 里用 class field 箭头 / useCallback / 构造器里绑一次。

new bound() 忽略 bind 的 this，保留参数

上文讲过。工程代码里"bind + new"组合少见，但用"工厂 + 预设参数"的写法会遇到。知道规则就行。

call(null, ...) 在非严格模式变 globalThis

非严格模式下 fn.call(null, args) 的 this 被"转换"为 globalThis（规范规定）。严格模式保留 null。ESM 自动严格 → 现代代码不用担心，但混严格/非严格场景要小心。

fn.call(0) 的装箱

非严格模式下 thisArg 是原始值会被装箱成 wrapper 对象。fn.call(1, ...) 里的 this 是 new Number(1)，typeof this === "object"。严格模式保留原始值。

bound function 的 .length / .name

bound.length 是"原函数长度 - 预设参数数"（下限 0）。bound.name 是 "bound " + original.name。调试时别惊讶。
